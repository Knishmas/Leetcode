class Solution {
    public int maxProfit(int[] prices) {
        
        //this is 1 # larger than the constraint of witch prices[i] could be
        int min = 10001;
        int maxProfit = 0;
        
        for(int i = 0; i < prices.length; i++){
            if(prices[i] < min){
                min = prices[i];
            }else if( (prices[i] - min) > maxProfit){
                maxProfit = prices[i] - min;
            }
        }
        return maxProfit;
    }
    
}
